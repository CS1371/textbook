% Constructing a cylinder
function main
    clear
    clc
    close all
    facets = 120;
    len = 2;
    radius = 0.5;
    [zz yy xx] = cylinder(radius, facets);
    xx = xx .* len;
    surf(xx, yy, zz);
    shading interp
    colormap copper
    axis equal
    xlabel('Z')
    ylabel('Y')
    zlabel('X')
    lightangle(60, 45)
    alpha(0.8)
    view(-20, 35)
end
