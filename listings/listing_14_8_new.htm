function play( score, instr, rate)
    global Fs;
    global music;
    Fs = 44100;
    tmax = 3; % sec
    N = tmax * Fs;
    t = (1:N) /Fs;
    % play middle C with a synthetic instrument
    freq = 261;
    w = 2 * pi * freq;
    note = zeros(1,N);
    for ndx = 1:length(instr)
        w = instr(ndx, 1) * 2000 * pi;
        rl = instr(ndx, 2);
        im = instr(ndx, 3);
        note = note + rl * cos(w * t);
        note = note + im * sin(w * t);
    end
    note = [note'; zeros(200000,1)];
    Ns = length(note);
    dt = 60/rate;
    where = 1;
    samples = Fs * dt;
    lngth = overall(score);
    maxSize = Ns + Fs * lngth * 60 / rate;
    music = zeros(maxSize,1);
    for ndx = 1:length(score)
        name = score{ndx, 1};
        dur = score{ndx, 2};
        pitch = getPitch(name);
        num = samples * dur;
        index = ceil(linspace(1, Ns, Ns/pitch));
        notes = note(index(1:num));
        music(where:(where+num-1)) = music(where:(where+num-1)) + notes;
        where = where + num;
    end
    music = music(music ~= 0);
    sound(music, Fs);
    pause(length(music) ./ Fs)
    audiowrite([inputname(1) '_' inputname(2) '.wav'], music, Fs )
end
function [pitch, muted] = getPitch(note)
    % Get the pitch of a note in ascii form
    half = 2.^(1/12);
    muted = false;
    switch note(1:end-1)
        case {'C'}
            power = 0;
        case {'R'}
            power = 0;
            muted = true;
        case {'CS' 'DF'}
            power = 1;
        case 'D'
            power = 2;
        case {'DS' 'EF'}
            power = 3;
        case 'E'
            power = 4;
        case 'F'
            power = 5;
        case {'FS' 'GF'}
            power = 6;
        case 'G'
            power = 7;
        case {'GS' 'AF'}
            power = 8;
        case 'A'
            power = 9;
        case {'AS' 'BF'}
            power = 10;
        case 'B'
            power = 11;
        otherwise
            error(['bad note value: ' note])
        end
        let = note(end) - '0';
        diff = let - 4;
        power = power + 12 * diff;
        pitch = half .^ power;
    end
function sum = overall(score)
    sum = 0;
    N = length(score);
    for it = 1:N
        sum = sum + score{it,2};
    end
